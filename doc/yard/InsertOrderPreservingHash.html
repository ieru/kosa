def merge!(other)
  new_keys = other.instance_variable_get(:@ordered_keys) || other.keys
  new_keys -= @ordered_keys
  @ordered_keys += new_keys
  super
  self
end

def reject!
  del = []
  each_pair {|k,v| del << k if yield k,v}
  del.each {|k| delete k}
  del.empty? ? nil : self
end
